Controllers

Um controller é uma função usada para 'aumentar' a instância do $scope. Para adicionar um valor inicial ou um comportamento para um objecto do $scope.


    function TodoCtrl($scope) {
      $scope.todos = [
        { text: 'Aprender AngularJS', done: true },
        { text: 'Criar um novo App', done: false }
      ];

      $scope.addTodo = function() {
        if ($scope.todoText) {
          $scope.todos.push({ text: $scope.todoText, done: false });
          $scope.todoText = '';
        }
      };
    };
